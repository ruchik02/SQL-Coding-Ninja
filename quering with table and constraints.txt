create table People(

table_name = 'People'
    ORDER BY

ID INT,
Customer_id INT,

Type VARCHAR(50) NOT NULL,

CREATE TABLE books (
  id int,
  title varchar(100) NOT NULL,
  author varchar(100) NOT NULL,
  published_date timestamp NOT NULL,
  isbn char(12),
  PRIMARY KEY (id),
  UNIQUE (isbn)
);

SELECT
   	table_name,
   	column_name,
   	data_type
	FROM
   	information_schema.columns
	WHERE
   	table_name = 'books';
    
 CREATE TABLE reviews (
  id int,
  book_id integer NOT NULL,
  reviewer_name varchar(255),
  content varchar(255),
  rating integer,
  published_date timestamp,
  PRIMARY KEY (id),
  FOREIGN KEY (book_id)
      REFERENCES books(id)
);

SELECT
   	table_name,
   	column_name,
   	data_type
	FROM
   	information_schema.columns
	WHERE
   	table_name = 'reviews';
